{
    	bool bCondition = true;
    	auto DecorateManager = ADecorateManager::GetDecorateManagerInstance();
    	if (DecorateManager->IsVideoMode())
    	{
        bCondition = GetVisibleInVideo();
        if(IsVideoAnimationForceSetVisibility != -1)
        {
            return IsVideoAnimationForceSetVisibility == 0 ？ false : true;
        }
    	}
    if (UElementList::IsElementListAvailable())
    {
        return UElementList::GetElementListInstance()->FusionActorVisiable(this) && bCondition;
    }
    return !GetHideStatus() && !GetIsDeleted() && bCondition;
}
